1.Código que substitua todas as palavras maiores que 5 caracteres por astericos (*).
# biblioteca
import re

# entrada de dados
txt = "Viver é acalentar sonhos e esperanças, fazendo da fé a nossa inspiração maior. É buscar nas pequenas coisas, um grande motivo para ser feliz."

# substitui os matches encontrados
x = re.sub(r'\b\w{6,}\b', '*', txt) # \b - qualquer limite de palavra
                                                     # \w - qualquer caráter não-palavra

# saida de dados
print(x)

3.Código que em um único REGEX, troque as letras maiúsculas por minúsculas, e vice-versa.
# biblioteca
import re

# entrada de dados
txt = "Viver é acalentar sonhos e esperanças, fazendo da fé a nossa inspiração maior. É buscar nas pequenas coisas, um grande motivo para ser feliz."

# substitui os matches encontrados
x = re.sub(r'[A-Za-zÁ-Úá-ú]', lambda x: x.group().lower() 
                                                  if x.group().isupper() 
                                                 else x.group().upper(), txt)

# saida de dados
print(x)
4.Código que retorna a posição inicial e final da maior palavra do texto.
# biblioteca
import re

# entrada de dados
txt = "Viver é acalentar sonhos e esperanças, fazendo da fé a nossa inspiração maior. É buscar nas pequenas coisas, um grande motivo para ser feliz."

# retornando as strings de todos os matches
x = re.finditer(r'\b\w+\b', txt)

maior = max(x, key=lambda x: len(x.group()))

# saida de dados - retorna uma tupla contendo a posição inicial e final de um
match
print(maior.span())
